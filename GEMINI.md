'' ).toLowerCase().replace(/\s+/g, '' ); return val.includes(normalized); }) ); }  Apply to every search bar. Pass relevant fields per context. 7. RECORD NUMBER GENERATION  // src/utils/recordNumbers.js import { db } from '../firebase/config' ; import { doc, runTransaction } from 'firebase/firestore' ; export async function generateRecordNumber(type) { const year = new Date ().getFullYear(); const prefix = { transfers: 'T' , quoteRequests: 'QR' , reconciliationReports: 'RC' }[type]; const counterRef = doc(db, 'recordCounters' , String (year)); const newNumber = await runTransaction(db, async (tx) =&gt; { const snap = await tx.get(counterRef); const current = snap.exists() ? (snap.data()[type] ?? 0 ) : 0 ; const next = current + 1 ; tx.set(counterRef, { [type]: next }, { merge: true }); return next; }); return `${prefix}#${year}-${String(newNumber).padStart(4, '0')}` ; }  8. SOFT DELETE SYSTEM Every delete must: Copy full document to trash collection Set isDeleted: true and deletedAt: now() on original Set purgeAt: now() + 30 days on trash document Log to auditLog Trigger any inventory rollbacks if needed All Firestore queries must include .where('isDeleted', '==', false) . Trash purge Cloud Function runs daily — permanently deletes documents where purgeAt &lt; now() . 9. INVENTORY UPDATE UTILITY All inventory mutations go through this single function:  // src/utils/inventoryUpdates.js import { db } from '../firebase/config' ; import { doc, runTransaction, collection, addDoc } from 'firebase/firestore' ; export async function adjustInventory({ skuId, location, delta, reason, relatedId, userId, userName }) { const invRef = doc(db, 'inventory' , `${skuId}_${location}` ); let oldQty, newQty; await runTransaction(db, async (tx) =&gt; { const snap = await tx.get(invRef); oldQty = snap.data().quantity; newQty = oldQty + delta; tx.update(invRef, { quantity: newQty }); }); await addDoc(collection(db, 'auditLog' ), { event: 'INVENTORY_ADJUSTED' , skuId, location, userId, userName, oldValue: oldQty, newValue: newQty, reason, relatedId, timestamp: new Date () }); }  Call from: manual adjustments, product pulls/undos, transfer receives, reconciliation submissions. Never write directly to inventory documents from components. 10. TRANSFER LOGIC Ship Transfer Set transfer.status = 'in_transit' For each item: set inventory.inTransitQty += requestedQty on &quot;from&quot; location (display only — do NOT call adjustInventory) Lock transfer (status !== 'requested' → hide all edit UI) Log to auditLog Trigger email notification Receive Transfer Compare receivedQty vs requestedQty per item If discrepancy: set transfer.discrepancyFlagged = true , log to auditLog, trigger discrepancy notification For each item: call adjustInventory() with positive delta on &quot;to&quot; location using receivedQty For each item: call adjustInventory() with negative delta on &quot;from&quot; location using receivedQty Clear inTransitQty on &quot;from&quot; inventory documents Set transfer.status = 'received' Log to auditLog Trigger email notification 11. SCHEDULED CLOUD FUNCTIONS  // functions/index.js const { onSchedule } = require( 'firebase-functions/v2/scheduler' ); const admin = require( 'firebase-admin' ); admin.initializeApp(); const db = admin.firestore(); // Daily: purge old trash + auto-delete discontinued SKUs at zero + check overdue counts exports.dailyMaintenance = onSchedule( 'every 24 hours' , async () =&gt; { const now = new Date (); // Purge trash older than 30 days const trash = await db.collection( 'trash' ).where( 'purgeAt' , '&lt;' , now).get(); trash.forEach(doc =&gt; doc.ref. delete ()); // Auto-delete discontinued SKUs at zero quantity const invSnap = await db.collection( 'inventory' ).get(); for ( const invDoc of invSnap.docs) { const inv = invDoc.data(); if (inv.quantity === 0 ) { const skuSnap = await db.collection( 'skus' ).doc(inv.skuId).get(); if (skuSnap.exists &amp;&amp; skuSnap.data().status === 'discontinued' &amp;&amp; !skuSnap.data().isDeleted) { await skuSnap.ref.update({ isDeleted: true, deletedAt: now }); await db.collection( 'auditLog' ).add({ event: 'SKU_AUTO_DELETED' , skuId: inv.skuId, sku: inv.sku, reason: 'Discontinued SKU reached zero quantity' , timestamp: now }); await triggerNotification( 'SKU_AUTO_DELETED' , { sku: inv.sku }); } } } // Check physical count overdue per location const settings = await db.collection( 'appSettings' ).doc( 'config' ).get(); const threshold = settings.data().physicalCountReminderDays || 35 ; const cutoff = new Date (now - threshold * 24 * 60 * 60 * 1000 ); const { lastFarmCount, lastMkeCount } = settings.data(); if (!lastFarmCount || lastFarmCount.toDate() &lt; cutoff) { await triggerNotification( 'PHYSICAL_COUNT_OVERDUE' , { location: 'Farm' }); } if (!lastMkeCount || lastMkeCount.toDate() &lt; cutoff) { await triggerNotification( 'PHYSICAL_COUNT_OVERDUE' , { location: 'MKE' }); } });  14.
